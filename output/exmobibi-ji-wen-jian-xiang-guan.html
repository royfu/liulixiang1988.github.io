文件下载与打开

$("a").click(function(){
    if($(this).attr('href').match(/doc$|docx$|pdf$/)){
        alert("matched");
    }
    ClientUtil.download($(this).attr('href'),"res:"+$(this).text(), null,null,false);
    NativeUtil.open("res:"+$(this).text());
});
